import numpy as np
import pandas as pd

# 
sigma_A = 0.05  # A
sigma_B = 0.12  # B
P_A = 2000000   # A
P_B = 1000000   # B
V = P_A + P_B   # 
w_A = P_A / V   # A
w_B = P_B / V   # B
z = 1.645       # 95%
rhos = [0.3, 0, -0.3]  # 

# 
def calculate_var(rho):
    sigma_p = np.sqrt(w_A**2 * sigma_A**2 + w_B**2 * sigma_B**2 + 2 * w_A * w_B * rho * sigma_A * sigma_B)
    VaR = z * sigma_p * V
    cov_A_p = w_A * sigma_A**2 + w_B * rho * sigma_A * sigma_B
    cov_B_p = w_A * rho * sigma_A * sigma_B + w_B * sigma_B**2
    MVaR_A = z * cov_A_p / sigma_p
    MVaR_B = z * cov_B_p / sigma_p
    CVaR_A = P_A * MVaR_A
    CVaR_B = P_B * MVaR_B
    return sigma_p, VaR, MVaR_A, MVaR_B, CVaR_A, CVaR_B

# 
results = []
for rho in rhos:
    sigma_p, VaR, MVaR_A, MVaR_B, CVaR_A, CVaR_B = calculate_var(rho)
    results.append({
        '': rho,
        '_p': round(sigma_p, 5),
        'VaR': int(VaR),
        'VaR_A ': MVaR_A ,
        'VaR_A': int(CVaR_A),
        'VaR_B ': MVaR_B ,
        'VaR_B': int(CVaR_B)
    })

# DataFrame
df = pd.DataFrame(results)

display(df)

# VaR = CVaR_A + CVaR_B
for index, row in df.iterrows():
    print(f" = {row['']}: VaR_A + VaR_B = {row['VaR_A'] + row['VaR_B']}   VaR = {row['VaR']} ")
